    <script>
        // MASSIV ERWEITERTE & 100% NEUTRALE NAMENS-POOLS
        const nickAdjectives = [
            "Silent", "Brave", "Fast", "Mighty", "Clever", "Fierce", "Noble", "Swift", "Wild", "Epic", 
            "Lucky", "Magic", "Cosmic", "Lunar", "Solar", "Crystal", "Golden", "Silver", "Iron", "Steel",
            "Happy", "Sunny", "Bright", "Calm", "Gentle", "Quick", "Smart", "Loyal", "Proud", "Bold",
            "Neon", "Cyber", "Rapid", "Stellar", "Astral", "Polar", "Aqua", "Terra", "Aero", "Nova"
        ];
        
        const nickNouns = [
            "Panther", "Eagle", "Lion", "Tiger", "Wolf", "Bear", "Falcon", "Shark", "Hawk", "Dragon", 
            "Phoenix", "Griffin", "Rhino", "Raven", "Owl", "Fox", "Stag", "Dolphin", "Whale", "Orca",
            "Manta", "Puma", "Jaguar", "Leopard", "Cheetah", "Lynx", "Bison", "Moose", "Elk", "Gazelle",
            "Zebra", "Hippo", "Elephant", "Mammoth", "Badger", "Otter", "Seal", "Penguin", "Pelican", "Toucan"
        ];

        // VERLÄNGERTE PASSWORT-POOLS (4 Kategorien)
        const passWords1 = ["blue", "red", "green", "yellow", "dark", "light", "cold", "warm", "deep", "high", "soft", "fast", "slow", "clear", "calm", "cool", "rich", "pure", "cyan", "teal", "mint"];
        const passWords2 = ["water", "fire", "earth", "wind", "stone", "wood", "metal", "sand", "snow", "rain", "ice", "sun", "moon", "star", "sky", "sea", "lake", "river", "hill", "tree", "leaf"];
        const passWords3 = ["apple", "mango", "berry", "lemon", "peach", "grape", "melon", "olive", "cedar", "maple", "birch", "pine", "oak", "palm", "fern", "moss", "rose", "lily", "tulip", "daisy"];
        const passWords4 = ["north", "south", "east", "west", "up", "down", "left", "right", "front", "back", "top", "base", "peak", "core", "edge", "line", "path", "road", "gate", "door", "wall"];

        function getRandomElement(arr) { return arr[Math.floor(Math.random() * arr.length)]; }

        async function generateAndSave() {
            const url = document.getElementById('supabaseUrl').value;
            const key = document.getElementById('supabaseKey').value;
            let siteUrl = document.getElementById('siteUrl').value;
            const count = parseInt(document.getElementById('userCount').value);
            const statusDiv = document.getElementById('status');
            const printArea = document.getElementById('print-area');

            if (!url || !key) {
                statusDiv.innerHTML = "<span class='text-red-500'>Error: Database credentials missing.</span>";
                return;
            }

            if (!siteUrl.endsWith('/')) {
                siteUrl += '/';
            }

            const supabaseClient = supabase.createClient(url, key);
            statusDiv.innerHTML = "<span class='text-stone-400'>Communicating with server...</span>";
            printArea.innerHTML = "";

            const playersToInsert = [];
            const usedNicknames = new Set();
            const usedPassphrases = new Set();

            while (playersToInsert.length < count) {
                // Nickname: Adjektiv + Tier + 3 Zahlen (z.B. CosmicDolphin842)
                const nickname = `${getRandomElement(nickAdjectives)}${getRandomElement(nickNouns)}${Math.floor(Math.random() * 900) + 100}`;
                
                // Passwort: 4 Wörter + 5 Zahlen für massive Sicherheit (z.B. blue-ocean-mango-north-73912)
                const passphrase = `${getRandomElement(passWords1)}-${getRandomElement(passWords2)}-${getRandomElement(passWords3)}-${getRandomElement(passWords4)}-${Math.floor(Math.random() * 90000) + 10000}`;

                if (!usedNicknames.has(nickname) && !usedPassphrases.has(passphrase)) {
                    usedNicknames.add(nickname);
                    usedPassphrases.add(passphrase);
                    playersToInsert.push({ passphrase: passphrase, nickname: nickname });
                }
            }

            const { data, error } = await supabaseClient.from('players').insert(playersToInsert).select();

            if (error) {
                statusDiv.innerHTML = `<span class='text-red-500'>System Error: ${error.message}</span>`;
                console.error(error);
                return;
            }

            statusDiv.innerHTML = `<span class='text-green-500'>Success: ${count} operatives deployed. Ready for print.</span>`;

            const chunkSize = 8;
            let globalIndex = 0;

            for (let i = 0; i < data.length; i += chunkSize) {
                const chunk = data.slice(i, i + chunkSize);
                
                const pageDiv = document.createElement('div');
                pageDiv.className = 'ticket-page';

                chunk.forEach((player) => {
                    const ticketDiv = document.createElement('div');
                    ticketDiv.className = 'ticket bg-stone-900 border border-stone-700 rounded-xl p-4 shadow-md relative';
                    
                    const loginUrl = `${siteUrl}?key=${player.passphrase}`;

                    ticketDiv.innerHTML = `
                        <div class="ticket-header w-full bg-stone-950 p-2 rounded-lg border border-stone-800 mb-3 flex justify-between items-center relative overflow-hidden">
                            <div class="absolute top-0 right-0 w-16 h-16 bg-green-900 rounded-full blur-2xl opacity-10 -mr-4 -mt-4 no-print"></div>
                            
                            <div class="relative z-10">
                                <h3 class="font-black text-lg tracking-tighter text-stone-100 uppercase leading-none">MN BG LTU: <span class="text-green-600">WorldCup '26</span></h3>
                                <p class="text-[9px] text-stone-500 font-bold tracking-widest uppercase mt-0.5">Prediction Game</p>
                            </div>
                            
                            <img src="Wappen.png" alt="Logo" class="w-10 h-10 object-contain drop-shadow-lg relative z-10 flex-shrink-0">
                        </div>

                        <p class="label-text">Username</p>
                        <p class="callsign text-lg font-black text-green-500 bg-stone-950 px-3 py-1 border border-stone-800 rounded-md shadow-inner mb-3">${player.nickname}</p>

                        <p class="label-text">Your Ticket to Glory</p>
                        <div id="qrcode-${globalIndex}" class="bg-white p-2 rounded-md shadow-sm border border-stone-200 mb-2 inline-block"></div>
                        
                        <p class="label-text w-full text-left">Manual Entry Link</p>
                        <p class="manual-link font-mono font-bold text-[8px] bg-stone-950 border border-stone-800 p-1.5 rounded w-full text-stone-300 break-all text-left shadow-inner mb-0">${loginUrl}</p>

                        <div class="instruction-box w-full mt-3 bg-stone-950 border border-stone-800 p-2 rounded-lg text-left shadow-inner">
                            <h4 class="font-black text-stone-200 uppercase tracking-widest text-[10px] border-b border-stone-800 pb-1 mb-1">Install as App</h4>
                            <div class="flex gap-2">
                                <div class="w-1/2">
                                    <p class="text-[9px] font-black text-stone-400 uppercase tracking-widest mb-1">iOS (Safari)</p>
                                    <ol class="text-[9px] text-stone-500 list-decimal pl-4 space-y-0.5">
                                        <li>Scan QR Code</li>
                                        <li>Tap Share Icon</li>
                                        <li>"Add to Home Screen"</li>
                                    </ol>
                                </div>
                                <div class="w-1/2 border-l border-stone-800 pl-2">
                                    <p class="text-[9px] font-black text-stone-400 uppercase tracking-widest mb-1">Android</p>
                                    <ol class="text-[9px] text-stone-500 list-decimal pl-3 space-y-0.5">
                                        <li>Scan QR Code</li>
                                        <li>Tap Menu (3 dots)</li>
                                        <li>"Add to Home screen"</li>
                                    </ol>
                                </div>
                            </div>
                        </div>
                    `;

                    pageDiv.appendChild(ticketDiv);
                    globalIndex++;
                });

                printArea.appendChild(pageDiv);
            }

            globalIndex = 0;
            data.forEach((player) => {
                const loginUrl = `${siteUrl}?key=${player.passphrase}`;
                new QRCode(document.getElementById(`qrcode-${globalIndex}`), {
                    text: loginUrl,
                    width: 80,
                    height: 80,
                    colorDark : "#000000",
                    colorLight : "#ffffff",
                    correctLevel : QRCode.CorrectLevel.M
                });
                globalIndex++;
            });
        }
    </script>
