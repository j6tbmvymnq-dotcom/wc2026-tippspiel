        async function saveBonusAnswers() {
            let hasError = false;
            for (const q of unansweredQuestions) {
                const val = document.getElementById(`answer-${q.id}`).value; 
                if (val !== '') {
                    // NEU: Jede Antwort wird sicher einzeln übergeben
                    const { error } = await supabaseClient.rpc('save_secure_bonus', { 
                        p_secret: currentPassphrase, 
                        p_question_id: q.id, 
                        p_answer: val 
                    });
                    if (error) hasError = true;
                }
            }
            
            if (hasError) {
                console.error("System Error saving Intel");
                alert("Fehler beim Sichern der Daten! Bitte lade die Seite neu.");
                return;
            }

            closeBonusModal();
            checkAndShowBonusQuestions(); 
        }
