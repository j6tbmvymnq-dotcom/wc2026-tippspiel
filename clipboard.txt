        let currentPassphrase = null; // NEU: Wir merken uns die Passphrase für spätere Speicher-Aktionen

        async function authenticateUser() {
            const urlParams = new URLSearchParams(window.location.search);
            const secretKey = urlParams.get('key');
            if (!secretKey) return document.getElementById('error-screen').classList.remove('hidden');
            
            // NEU: Wir nutzen den sicheren RPC-Login
            const { data: players, error } = await supabaseClient.rpc('login_operative', { p_secret: secretKey });
            
            if (error || !players || players.length === 0) return document.getElementById('error-screen').classList.remove('hidden');
            
            currentPassphrase = secretKey; // Passphrase für die Session speichern
            loadDashboard(players[0]); // Das Array entpacken
        }
