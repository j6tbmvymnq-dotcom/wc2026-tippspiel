    <script>
        const nickAdjectives = [
            "Silent", "Brave", "Fast", "Mighty", "Clever", "Fierce", "Noble", "Swift", "Wild", "Epic", 
            "Lucky", "Magic", "Cosmic", "Lunar", "Solar", "Crystal", "Golden", "Silver", "Iron", "Steel",
            "Happy", "Sunny", "Bright", "Calm", "Gentle", "Quick", "Smart", "Loyal", "Proud", "Bold",
            "Neon", "Cyber", "Rapid", "Stellar", "Astral", "Polar", "Aqua", "Terra", "Aero", "Nova"
        ];
        
        const nickNouns = [
            "Panther", "Eagle", "Lion", "Tiger", "Wolf", "Bear", "Falcon", "Shark", "Hawk", "Dragon", 
            "Phoenix", "Griffin", "Rhino", "Raven", "Owl", "Fox", "Stag", "Dolphin", "Whale", "Orca",
            "Manta", "Puma", "Jaguar", "Leopard", "Cheetah", "Lynx", "Bison", "Moose", "Elk", "Gazelle",
            "Zebra", "Hippo", "Elephant", "Mammoth", "Badger", "Otter", "Seal", "Penguin", "Pelican", "Toucan"
        ];

        const passWords1 = ["blue", "red", "green", "yellow", "dark", "light", "cold", "warm", "deep", "high", "soft", "fast", "slow", "clear", "calm", "cool", "rich", "pure", "cyan", "teal", "mint"];
        const passWords2 = ["water", "fire", "earth", "wind", "stone", "wood", "metal", "sand", "snow", "rain", "ice", "sun", "moon", "star", "sky", "sea", "lake", "river", "hill", "tree", "leaf"];
        const passWords3 = ["apple", "mango", "berry", "lemon", "peach", "grape", "melon", "olive", "cedar", "maple", "birch", "pine", "oak", "palm", "fern", "moss", "rose", "lily", "tulip", "daisy"];
        const passWords4 = ["north", "south", "east", "west", "up", "down", "left", "right", "front", "back", "top", "base", "peak", "core", "edge", "line", "path", "road", "gate", "door", "wall"];

        function getRandomElement(arr) { return arr[Math.floor(Math.random() * arr.length)]; }

        // Globale Variablen für den Druck
        let deployedOperatives = [];
        let deploySiteUrl = "";

        async function generateAndSave() {
            const url = document.getElementById('supabaseUrl').value;
            const key = document.getElementById('supabaseKey').value;
            deploySiteUrl = document.getElementById('siteUrl').value;
            const count = parseInt(document.getElementById('userCount').value);
            const statusDiv = document.getElementById('status');
            const logDiv = document.getElementById('deployment-log');

            if (!url || !key) {
                statusDiv.innerHTML = "<span class='text-red-500'>Error: Database credentials missing.</span>";
                return;
            }

            if (!deploySiteUrl.endsWith('/')) {
                deploySiteUrl += '/';
            }

            const supabaseClient = supabase.createClient(url, key);
            statusDiv.innerHTML = "<span class='text-stone-400'>Communicating with server...</span>";
            logDiv.innerHTML = "";
            document.getElementById('printBtn').disabled = true;

            const playersToInsert = [];
            const usedNicknames = new Set();
            const usedPassphrases = new Set();

            while (playersToInsert.length < count) {
                const nickname = `${getRandomElement(nickAdjectives)}${getRandomElement(nickNouns)}${Math.floor(Math.random() * 900) + 100}`;
                const passphrase = `${getRandomElement(passWords1)}-${getRandomElement(passWords2)}-${getRandomElement(passWords3)}-${getRandomElement(passWords4)}-${Math.floor(Math.random() * 90000) + 10000}`;

                if (!usedNicknames.has(nickname) && !usedPassphrases.has(passphrase)) {
                    usedNicknames.add(nickname);
                    usedPassphrases.add(passphrase);
                    playersToInsert.push({ passphrase: passphrase, nickname: nickname });
                }
            }

            const { data, error } = await supabaseClient.from('players').insert(playersToInsert).select();

            if (error) {
                statusDiv.innerHTML = `<span class='text-red-500'>System Error: ${error.message}</span>`;
                console.error(error);
                return;
            }

            // Speichern in globaler Variable für den Druck
            deployedOperatives = data;

            statusDiv.innerHTML = `<span class='text-green-500'>Success: ${count} operatives deployed. Ready for print.</span>`;
            
            // Log anzeigen
            data.forEach(p => {
                logDiv.innerHTML += `<div class="bg-stone-900 p-2 rounded border border-stone-800 text-center">${p.nickname}</div>`;
            });

            // Druck-Button aktivieren
            const printBtn = document.getElementById('printBtn');
            printBtn.disabled = false;
            printBtn.className = "bg-green-700 text-white font-black py-4 px-4 rounded-lg border border-green-600 hover:bg-green-600 transition-colors shadow-lg uppercase tracking-widest text-sm flex justify-center items-center gap-2 cursor-pointer";
        }

        // DIE NEUE, SAUBERE DRUCK-FUNKTION
        // ==========================================
        function openPrintWindow() {
            if(deployedOperatives.length === 0) return;

            // Ermittelt den genauen Pfad zum Bild, damit es im neuen Fenster sicher lädt
            const baseUrl = window.location.href.substring(0, window.location.href.lastIndexOf('/') + 1);
            const wappenSrc = `${baseUrl}Wappen.png`;

            // Öffnet einen komplett neuen Tab
            const printWin = window.open('', '_blank');
            
            // HTML-Gerüst für das Druckfenster aufbauen
            let html = `
            <!DOCTYPE html>
            <html lang="en">
            <head>
                <meta charset="UTF-8">
                <title>Print Tickets - WC26</title>
                <script src="https://cdnjs.cloudflare.com/ajax/libs/qrcodejs/1.0.0/qrcode.min.js"><\/script>
                <style>
                    /* SAUBERES DRUCK-CSS: Keine Hacks, nur reines Grid-Layout */
                    @page { margin: 0; size: A4 portrait; }
                    body { 
                        margin: 0; padding: 0; 
                        background: white; 
                        font-family: system-ui, -apple-system, sans-serif; 
                        -webkit-print-color-adjust: exact !important; 
                        print-color-adjust: exact !important; 
                    }
                    
                    /* Eine Seite ist exakt ein DIN A4 Blatt (4 Tickets) */
                    .page {
                        width: 210mm; 
                        height: 297mm;
                        padding: 12mm;
                        box-sizing: border-box;
                        display: grid;
                        grid-template-columns: 1fr 1fr;

                        page-break-after: always;
                    }

                    /* Das 4er-Layout erlaubt viel größere und schönere Tickets */
                    .ticket {

                    }

                    .label { font-size: 10px; font-weight: 800; color: #57534e; text-transform: uppercase; letter-spacing: 1.5px; margin-bottom: 6px; }
                    
                    .username { background: #f0fdf4; color: #166534; border: 2px solid #bbf7d0; font-size: 20px; font-weight: 900; padding: 10px; border-radius: 6px; text-align: center; margin-bottom: 20px; }
                    
                    .qr-wrapper { display: flex; justify-content: center; margin-bottom: 15px; }
                    .qr-container { padding: 4px; background: white; border: 1px solid #d6d3d1; border-radius: 6px; }
                    
                    .link-box { background: #f5f5f4; border: 1px dashed #a8a29e; font-family: monospace; font-size: 11px; padding: 10px; border-radius: 6px; color: #292524; word-break: break-all; margin-bottom: auto; text-align: center; font-weight: bold;}
                    
                    .instructions { background: #f5f5f4; border: 2px solid #d6d3d1; padding: 12px; border-radius: 8px; margin-top: 20px; }
                    .instructions h4 { margin: 0 0 8px 0; font-size: 11px; text-transform: uppercase; border-bottom: 1px solid #d6d3d1; padding-bottom: 4px; color: #1c1917; font-weight: 900;}
                    .inst-cols { display: flex; gap: 10px; }
                    .inst-col { flex: 1; }
                    .inst-col p { margin: 0 0 4px 0; font-size: 10px; font-weight: 900; text-transform: uppercase; color: #57534e; }

                </style>
            </head>
            <body>
            `;

            // Tickets generieren (in 4er-Blöcken)
            const chunkSize = 4;
            let qrIndex = 0;
            
            for(let i=0; i < deployedOperatives.length; i += chunkSize) {
                const chunk = deployedOperatives.slice(i, i + chunkSize);
                
                html += `<div class="page">`; // Starte A4 Seite
                
                chunk.forEach(player => {
                    const loginUrl = `${deploySiteUrl}?key=${player.passphrase}`;
                    html += `
                        <div class="ticket">
                            <div class="header">
                                <h3>MN BG LTU: <span>WorldCup '26</span></h3>

                            <div class="label">Username</div>
                            <div class="username">${player.nickname}</div>
                            
                            <div class="label" style="text-align: center;">Your Ticket to Glory</div>
                            <div class="qr-wrapper">
                                <div class="qr-container" id="qr-${qrIndex}" data-url="${loginUrl}"></div>

                                            <li>"Add to Home screen"</li>

                                    </div>
                                </div>
                            </div>
                        </div>
                    `;
                    qrIndex++;
                });
                
                html += `</div>`; // Beende A4 Seite
            }

            // Skript hinzufügen, das im neuen Fenster die QR-Codes zeichnet und dann druckt
            html += `
                <script>
                    window.onload = function() {
                        const qrs = document.querySelectorAll('.qr-container');
                        qrs.forEach(container => {
                            new QRCode(container, {
                                text: container.getAttribute('data-url'),
                                width: 140, // Schön groß für die 4er Ansicht
                                height: 140,
                                colorDark : "#000000",
                                colorLight : "#ffffff",
                                correctLevel : QRCode.CorrectLevel.M
                            });
                        });
                        
                        // Gibt dem Browser 0.5 Sekunden Zeit, die Codes und das Wappen zu laden, dann öffnet sich das Druckmenü
                        setTimeout(() => {
                            window.print();
                        }, 500);
                    };
                <\/script>
            </body>
            </html>
            `;

            // Den fertigen Code in den neuen Tab schreiben
            printWin.document.write(html);
            printWin.document.close();
        }
    </script>
